远程登录服务器

  1. 配置root及应用账号权限

    sudo adduser kingdee
Enter new UNIX password: 
sudo gpasswd -a kingdee sudo //赋予用户权限
sudo visudo

    // User privilege specification
    root ALL=(ALL:ALL) ALL
    kingdee ALL=(ALL:ALL) ALL

    service ssh restart

  1. 配置ssh无密码登录
    ssh-keygen -t rsa -b 4096 -C "dongwenzhao@lngtop.com" //生成秘钥
eval "$(ssh-agent -s)" 
ssh-add ~/.ssh/id_rsa
cd ~/.ssh/   //进到ssh文件夹
vi  authorized_keys
//拷贝本地的rsa_pub  到服务器
chmod 600 authorized_keys //修改权限
sudo service ssh restart
  1. 修改服务器默认端口
    sudo  vi /etc/ssh/sshd_config

# What ports, IPs and protocols we listen for
Port 22 //修改端口
AllowUsers kingdee //添加用户
sudo service ssh restart //重启

sudo vi /etc/ssh/sshd_config
 PermitRootLogin no  //关闭root登录
sudo service ssh restart //重启
